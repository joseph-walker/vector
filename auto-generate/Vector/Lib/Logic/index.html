andCombinator

Source

Logical And Combinator :: [(a -> Bool)] -> a -> Bool

Given n functions {f1, f2, ..., fn}, combine them in such a way to produce a new function g that returns true given {f1(x), f2(x), ... fn(x)} all return true.

$funcF = function($x) { return $x < 5; };
$funcG = function($x) { return $x > 0; };
$combinator = Logic::andCombinator([$funcF, $funcG]);
$combinator(4); // True
$combinator(2); // True
$combinator(7); // False

orCombinator

Source

Logical Or Combinator :: [(a -> Bool)] -> a -> Bool

Given n functions {f1, f2, ..., fn}, combine them in such a way to produce a new function g that returns true given at least one of {f1(x), f2(x), ... fn(x)} return true.

$funcF = function($x) { return $x >= 5; };
$funcG = function($x) { return $x == 0; };
$combinator = Logic::orCombinator([$funcF, $funcG]);
$combinator(9); // True
$combinator(0); // True
$combinator(2); // False
